    agent any
    options {
        skipStagesAfterUnstable()
    }
    stages {
        stage('Building') {
            steps {
                sh 'make'
            }
        }
        stage('Testing'){
            steps {
                sh 'make check'

            }
        }
        stage('Deploying') {
            steps {
                sh 'make publish'
            }
        }
    }
}
